def linear(x,a,b):
    return a*x+b

def quad(x,a,b,c):
    return a*(x**2)+b*x+c

def cubic(x,a,b,c,d):
    return a*(x**3)+b*(x**2)+c*x+d

def log(x,a,b):
    return a*np.log(b+x)

def power(x,a,b):
    return a*(x**b)

def expo(x,a,b,c):
    return a*np.exp(b*x)+c

def log2(x,a,b):
    return a*np.log(x)+b
